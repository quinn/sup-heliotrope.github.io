+  def shell_out_async editor, file
+    @shelled = true
+    command = "#{editor} #{file.path}"
+    mtime = File.mtime file.path
+
+    Ncurses.sync do
+      pid = Process.fork
+      if pid.nil?
+        exec(command)
+      else
+        pthread = Process.detach(pid)
+        PollManager.editor_loop_init pthread, file, mtime
+      end
+      Ncurses.stdscr.keypad 1
+      Ncurses.refresh
+      Ncurses.curs_set 0
+    end
+    @shelled = false
+  end
