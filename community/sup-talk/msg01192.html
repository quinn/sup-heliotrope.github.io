# http://sup.rubyforge.org/wiki/wiki.pl?Hooks
def offlineimap()
  cmd = "offlineimap -o -l ~/.offlineimap.log"
  `#{cmd} 2>&1`
end

say "Checking if we can run offlineimap..."
if (@last_fetch || Time.at(0)) < Time.now - 120
  say "Running offlineimap..."
  log offlineimap()
